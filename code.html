                        // get the values from the UI
                        function getValues(){
                            // get the values from the page 
                            let startValue = document.getElementById("startValue").value;
                            let endValue = document.getElementById("endValue").value;
                            let numbers = [];

                            // parse into integers
                            startValue = parseInt(startValue);
                            endValue = parseInt(endValue);

                            if(Number.isInteger(startValue) && Number.isInteger(endValue)) {
                                // call generateNumbers 
                                numbers = generateNumbers(startValue, endValue);

                                // call displayNumbers
                                displayNumbers(numbers);
                            } else {
                                alert("You must enter only numbers.");
                            }
                        }

                        // generate numbers from the start value, to the end value
                        function generateNumbers(startValue, endValue) {
                            let numbers = [];

                            // we want to get all numbers from start to end

                            for(let i = startValue; i <= endValue; i++) {

                                // this will execute in a loop, until index = endValue
                                numbers.push(i);
                            }

                            return numbers;
                        }

                        // display the numbers and mark the even numbers bold
                        function displayNumbers(numbers) {

                            let templateRows = "";

                            for (let i = 0; i < numbers.length; i++) {
                                
                                let number = numbers[i];
                                if (number % 2 == 0) {
                                    className="even"
                                } else {
                                    className="odd"
                                }
                                // This does render correctly with Prism. See the source.
                                templateRows += `${number}`;
                            }

                            document.getElementById("results").innerHTML = templateRows;
                        }

The code is structured into the three following functions.

Get Values

Get Values is the function that initializes the Zero to Hundred application. It does this by grabbing the values entered into the form. Once that is done, it parses the data received from the form, to ensure it is an integer. If not, an error is triggered, and the user is asked to enter the proper type of values.

If the numbers are valid, Generate Numbers is called. Once this is successfully completed, Display Numbers is also called, and the numbers array generated by Generate Numbers is passed into it.

Generate Numbers

This function takes the start and end values selected by the user, and generates an array of numbers. That array is returned to Get Values, which is then received by Display Numbers.

Display Numbers

This function creates a table around the generated numbers array. Once the array of numbers has been passed through to Display Numbers, the for loop within iterates over each entry in the array, and creates a column and row for each number in the table.

As each cell in the table is being created, the value is being checked whether it is even or odd. If the number is even, the font weight of that item is changed to bold. Once this check is completed, the table and it's rows are added to the templateRows variable. The Document's method Get Element by Id is attached to the templateRows variable, and then displayed on the DOM.
